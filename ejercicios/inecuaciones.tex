\question Demuestra que son equivalentes las siguientes inecuaciones:
\begin{multicols}{3}
\begin{parts}
\part[] $\dfrac{x}{3} - 3x + 4 > x - 2$   
\part[]  $7x < 12$ 
\begin{solution} No son equivalentes: $\left(-\infty, \frac{18}{11}\right) \neq \left(-\infty, \frac{12}{7}\right)$   \end{solution}
\end{parts}
\end{multicols}

\question ¿Son equivalentes las siguientes inecuaciones?
\begin{multicols}{3}
\begin{parts}
\part[]  $x - 2 \geqslant \frac{x}{3}$ 
\part[]  $3x - \frac{x}{6} + 2 \geqslant 3 + \frac{{5x}}{2}$
\begin{solution} Sí: $\left[3, \infty\right)=\left[3, \infty\right)$ \end{solution}
\end{parts}
\end{multicols}
