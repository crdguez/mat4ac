\question Resuelve las siguientes inecuaciones:
\begin{multicols}{2}
\begin{parts}
\part[1] $\frac{x^{2} - 3 x + 2}{x^{2} - 1} \leq 0$
\begin{solution}
$\left(-1, 1\right) \cup \left(1, 2\right]$
\end{solution}
\part[1] $\frac{x^{2} + x - 2}{x^{2} - 4} \geq  0 $
\begin{solution}
$\left(-\infty, -2\right) \cup \left(-2, 1\right] \cup \left(2, \infty\right)$
\end{solution}
\part[1] $\frac{x^{2} - 3 x + 2}{x^{2} + x - 2} \leq 0$
\begin{solution}
$\left(-2, 1\right) \cup \left(1, 2\right]$
\end{solution}

\end{parts}

\end{multicols}
